# linux 通用常用命令

# 设置 root 密码

ubuntu centos
sudo passwd

# 切换到 root

ubuntu centos
sudo su

# 删除

  • 此命令慎用
  • -rf后接文件或文件夹
sudo rm -rf /opt/dotnet/sdk/1.0.4

# 软链接

# 创建

sudo ln -s /src /src-ln

# 删除软链接

  • 注意，/src-ln/多加个斜杠会把/src里面的内容也干掉
sudo rm -rf /src-ln

# 查看 cpu、进程

top

# 查看内存

free -h

# 环境变量

  • export DOCKER_HOST=tcp://127.0.0.1:2375
  • 持久化
    • 执行vi ~/.bashrc编辑此文件，末尾处添加export DOCKER_HOST=tcp://127.0.0.1:2375，然后重启即可生效

# 文件权限

  • sudo chown -R username dirname
  • 修改文件夹所属用户
